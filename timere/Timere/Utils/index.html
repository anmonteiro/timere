Search oriented operations

Warning: These are low level operations used to communicate with the resolver directly.

While powerful and necessary for certain queries, may yield unexpected results.

val follow : Duration.t -> t -> t -> t

follow bound t1 t2, let s1 = resolve t1, s2 = resolve t2 for every interval (x1, y1) in s1, search for (x2, y2) in s2 up to x2 <= x1 + bound, and return (x2, y2) if found

val between_inc : Duration.t -> t -> t -> t

between_inc bound s1 s2, let s1 = resolve t1, s2 = resolve t2 for every interval (x1, y1) in s1, search for (x2, y2) in s2 up to x2 <= x1 + bound, and return (x1, y2) if found

val between_exc : Duration.t -> t -> t -> t

between_inc bound s1 s2, let s1 = resolve t1, s2 = resolve t2 for every interval (x1, y1) in s1, search for (x2, y2) in s2 up to x2 <= x1 + bound, and return (x1, x2) if found
